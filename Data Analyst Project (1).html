# Top Descriptors by Variance
import pandas as pd

# List of numeric columns
numeric_columns = ['MolWt', 'TPSA', 'BalabanJ', 'Chi0', 'Dimention','BertzCT']

# Initialize an empty list for results
results = []

# Loop through columns to calculate variance
for column in numeric_columns:
    query = f"""
    SELECT '{column}' AS Descriptor, VAR_POP({column}) AS Variance
    FROM drug_descriptors
    WHERE Activity = 'Active'
    """
    df_variance = pd.read_sql(query, conn)
    results.append(df_variance)

# Combine results into a single DataFrame
combined_results = pd.concat(results, ignore_index=True)

# Convert Variance to numeric to avoid sorting issues
combined_results['Variance'] = pd.to_numeric(combined_results['Variance'], errors='coerce')

# Sort by variance in ascending or descending order
top_descriptors = combined_results.sort_values(by='Variance', ascending=False)  # Change to False for descending

# Reset the index to have sequential numbering (0, 1, 2, ...)
top_descriptors = top_descriptors.reset_index(drop=True)

# Display the results
print("Top Descriptors by Variance:")
print(top_descriptors)
